5. (12pt)

one : .data

7. 실행결과를 쓰시오 (12pt)
pid_t pid;

void f(void)
{
	printf("{1}\n");
}

void g(void)
{
	printf("{2}\n");
}

void h(int sig)
{
	printf("{3}\n");
	kill(pid, SIGUSR1);
}

void i(int sig)
{
	printf("{4}\n");
	exit (0);
}

void main() {
	signal(SIGUSR1, h);
	atexit(f);
	
	printf("{5}\n");
	
	if ((pid = fork()) == 0) {
		signal(SIGUSR1, i);
		kill(getppid(), SIGUSR1);
		while (1) {};
	} else {
		atexit(g);
		waitpid(pid, NULL, 0);
	}
	printf("{6}\n");
}
